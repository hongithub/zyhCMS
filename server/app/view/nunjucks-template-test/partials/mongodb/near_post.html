5. 实际使用场景展示

在文章详情页的使用方法:
<!-- 在文章详情页模板中使用 -->
{% raw %}{% nearpost key="navigation", id="{{ post.id }}" %}{% endraw %}
{% raw %}{% if navigation %}{% endraw %}
  <nav class="post-navigation">
    {% raw %}{% if navigation.prePost %}{% endraw %}
      <div class="nav-prev">
        <a href="/details/{% raw %}{{ navigation.prePost.id }}{% endraw %}.html">
          上一篇: {% raw %}{{ navigation.prePost.title }}{% endraw %}
        </a>
      </div>
    {% raw %}{% endif %}{% endraw %}
    
    {% raw %}{% if navigation.nextPost %}{% endraw %}
      <div class="nav-next">
        <a href="/details/{% raw %}{{ navigation.nextPost.id }}{% endraw %}.html">
          下一篇: {% raw %}{{ navigation.nextPost.title }}{% endraw %}
        </a>
      </div>
    {% raw %}{% endif %}{% endraw %}
  </nav>
{% raw %}{% endif %}{% endraw %}
